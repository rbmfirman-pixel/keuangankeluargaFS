const API_URL = "https://script.google.com/macros/s/AKfycbzQElXfKhIb5ORqq2Q71iZeOWMKil3oLLBaCOkF0-ZRRJqpRHKrhDKtXniqVpC2BJYH/exec"; // Tambahkan variabel ini di atas script let editMode = false; let editRow = null; // FUNGSI BARU UNTUK INPUT/EDIT function openInputMode() { editMode = false; document.getElementById('form-title').innerText = "Input Transaksi"; document.getElementById('transaksi-form').reset(); document.getElementById('in-pencatat').value = localStorage.getItem("fw_user") || "Admin"; switchTab('input'); } function openEdit(item) { editMode = true; editRow = item.row; document.getElementById('form-title').innerText = "Edit Transaksi"; document.getElementById('in-pencatat').value = item.pencatat; document.getElementById('in-jenis').value = item.jenis; document.getElementById('in-tgl').value = new Date(item.tgl).toISOString().split('T')[0]; document.getElementById('in-kategori').value = item.kategori; document.getElementById('in-keterangan').value = item.keterangan; document.getElementById('in-nominal').value = new Intl.NumberFormat('id-ID').format(item.nominal); switchTab('input'); } function formatRupiah(input) { let val = input.value.replace(/[^0-9]/g, ''); input.value = val ? new Intl.NumberFormat('id-ID').format(val) : ''; } function confirmSubmit() { if(confirm("Yakin ingin menyimpan?")) submitForm(); } function submitForm() { const payload = { action: 'saveTransaction', mode: editMode ? 'edit' : 'add', row: editRow, user: localStorage.getItem("fw_user"), pencatat: document.getElementById('in-pencatat').value, jenis: document.getElementById('in-jenis').value, tgl: document.getElementById('in-tgl').value, kategori: document.getElementById('in-kategori').value, keterangan: document.getElementById('in-keterangan').value, nominal: parseInt(document.getElementById('in-nominal').value.replace(/\./g, '')) }; fetch(API_URL, { method: 'POST', body: JSON.stringify(payload) }) .then(res => res.json()).then(data => { alert('Berhasil!'); location.reload(); }); } // Update fungsi switchTab agar support tab 'input' function switchTab(tab) { ['home', 'riwayat', 'laporan', 'input'].forEach(v => { document.getElementById(`view-${v}`)?.classList.toggle('hidden', v !== tab); }); // ... sisa logika tab lainnya }
